document.getElementById('host-modal').style.display='flex'; // --- Host Logic --- document.getElementById('start-host-btn').onclick = () => { const name = document.getElementById('host-name-input').value; const pin = document.getElementById('host-pin-input').value; if(name && pin.length === 6) { isHost = true; startPeerSession(pin, name); } else { alert("Enter Name and 6-Digit PIN"); } }; // --- Join Logic --- document.getElementById('join-btn').onclick = () => { const pin = document.getElementById('pin-entry').value; if(pin.length === 6) { isHost = false; startPeerSession(null, null, pin); } }; function startPeerSession(pin, roomName, joinPin) { const targetPin = isHost ? pin : null; const peerId = targetPin ? PROJECT_PREFIX + targetPin : null; // Create the peer connection peer = new Peer(peerId, peerConfig); peer.on('open', (id) => { console.log('My peer ID is: ' + id); if (isHost) { setupRoomUI(roomName, pin); } else { connectToHost(joinPin); } }); peer.on('connection', (c) => { c.on('open', () => { if (isHost) { connections.push(c); c.send({type: 'room-info', name: roomName, pin: pin}); appendMessage("System", "#777", "Someone joined the room."); } }); c.on('data', (data) => handleData(data, c)); }); peer.on('error', (err) => { console.error("PeerJS Error:", err.type); if (err.type === 'unavailable-id') { alert("This PIN is already hosting a room!"); } else { showPinError(); } }); } function connectToHost(pin) { // Try to connect to the host's specific ID conn = peer.connect(PROJECT_PREFIX + pin, { reliable: true }); conn.on('open', () => { console.log("Connected to host"); conn.send({type: 'join-request', name: myName, color: myColor}); }); conn.on('data', handleData); // Timeout if host doesn't respond setTimeout(() => { if (!conn.open) { showPinError(); peer.destroy(); } }, 5000); } function handleData(data, senderConn) { if (data.type === 'msg') appendMessage(data.name, data.color, data.text); if (data.type === 'room-info') setupRoomUI(data.name, data.pin); if (data.type === 'disband') { alert("Room Disbanded"); location.reload(); } if (data.type === 'kick') { alert("You have been kicked"); location.reload(); } // Host-specific handling if (isHost && data.type === 'join-request') { appendMessage("System", "#777", data.name + " has entered the chat."); } } function appendMessage(name, color, text) { const msgArea = document.getElementById('messages'); const div = document.createElement('div'); div.className = "chat-msg"; div.style.borderColor = color; div.style.alignSelf = (name === myName) ? "flex-end" : "flex-start"; div.innerHTML = `
${name}
${text}
`; msgArea.appendChild(div); msgArea.scrollTop = msgArea.scrollHeight; } document.getElementById('send-btn').onclick = sendMessage; document.getElementById('msg-input').onkeypress = (e) => { if(e.key === 'Enter') sendMessage(); }; function sendMessage() { const input = document.getElementById('msg-input'); const val = input.value.trim(); if(!val) return; appendMessage(myName, myColor, val); const packet = {type: 'msg', name: myName, color: myColor, text: val}; if(isHost) { connections.forEach(c => { if(c.open) c.send(packet); }); } else if(conn && conn.open) { conn.send(packet); } input.value = ""; } function showPinError() { document.getElementById('pin-entry').style.color = "#f44336"; document.getElementById('pin-error').style.display = "block"; } document.getElementById('disband-btn').onclick = () => { if(confirm("Disband room?")) { connections.forEach(c => { if(c.open) c.send({type: 'disband'}); }); setTimeout(() => location.reload(), 500); } }; document.getElementById('pin-entry').oninput = () => { document.getElementById('pin-entry').style.color = "white"; document.getElementById('pin-error').style.display = "none"; }; function setupRoomUI(name, pin) { document.getElementById('chat-home').style.display = 'none'; document.getElementById('host-modal').style.display = 'none'; document.getElementById('chat-room').style.display = 'flex'; document.getElementById('room-header-info').style.display = 'block'; document.getElementById('display-room-name').textContent = name; document.getElementById('display-pin').textContent = "PIN: " + pin; if(isHost) document.getElementById('disband-btn').style.display = 'block'; }
